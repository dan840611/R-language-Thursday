URL   = "http://news.cnyes.com/news/cat/headline"
html  = read_html(URL)
title = html_nodes(html, "a")
href  = html_attr(title, "href")
data = data.frame(title = toUTF8(html_text(title)),
                    href = href)
data = data[-c(1:106,138,117),]
  
getContent <- function(x) {
    url  = paste0("http://news.cnyes.com/", x)
    tag  = html_node(read_html(url), 'section._82F')
    text = toUTF8(html_text(tag))
}

allText = sapply(data$href, getContent)
write.table(allText, "cnYES.txt")

去除掉無意義的字詞，並因為文章數僅有30篇，故將詞彙頻率降低，畫出文字雲。

library(NLP)
library(tm)

filenames <- list.files(getwd(), pattern="*.txt")
files <- lapply(filenames, readLines)
docs <- Corpus(VectorSource(files))
#移除可能有問題的符號
toSpace <- content_transformer(function(x, pattern) {
  return (gsub(pattern, " ", x))
}
)
docs <- tm_map(docs, toSpace, "\n\n")

#移除標點符號 (punctuation)
#移除數字 (digits)、空白 (white space)
docs <- tm_map(docs, removePunctuation)
docs <- tm_map(docs, removeNumbers)
docs <- tm_map(docs, stripWhitespace)
#語詞詞幹化 (stemmization)
#以英文為例
#https://zh.wikipedia.org/wiki/%E8%AF%8D%E5%B9%B2%E6%8F%90%E5%8F%96
#library(SnowballC)
#確保任何形式的單字只會轉換成相同詞性出現一次
#docs <- tm_map(docs, stemDocument)
mixseg = worker()
jieba_tokenizer=function(d){
  unlist(segment(d[[1]],mixseg))
}
seg = lapply(docs, jieba_tokenizer)
freqFrame = as.data.frame(table(unlist(seg)))

透過文字雲可以發現，中國、UA、美國及成長等出現頻率相對於其他字來得多。以目前國際經濟來看，中國及美國仍是影響世界經濟的主要兩大經濟體，因此在鉅亨網上出現次數多，是可以預見的，另外「成長」出現的次數也多，從今年各國股市來看，確實也都呈現成長，包括諸多國家的股市均創新高，台股也站穩萬點，因此使得財經文章「成長」出現多次。
